<script>
let newEl;
function createEl() {
  newEl = document.createElement("p");
  newEl.textContent = "New paragraph added!";
}
function appendEl() {
  if(newEl) document.getElementById("box").appendChild(newEl);
}
function insertEl() {
  let p = document.createElement("p");
  p.textContent = "Inserted at top!";
  document.getElementById("box").insertAdjacentElement("afterbegin", p);
}
function removeEl() {
  let box = document.getElementById("box");
  if(box.lastChild) box.removeChild(box.lastChild);
}
</script>

function logRequests(req, res, next) {
  const method = req.method;
  const url = req.url;
  const timestamp = new Date().toISOString();

  console.log(`[${timestamp}] ${method} ${url}`);
  next(); // Pass control to next middleware or route handler
}

// Apply middleware globally
app.use(logRequests);

// Example route
app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(PORT, () => {
  console.log(`Server running at http://localhost:${PORT}`);

SERVER.JS
const express = require('express');
const fs = require('fs');
const fsPromises = require('fs').promises;
const app = express();
const PORT = 3000;

// Serve static HTML file
app.use(express.static(__dirname));

// Callback method
app.get('/read-callback', (req, res) => {
  fs.readFile('sample.txt', 'utf8', (err, data) => {
    if (err) res.send('Error reading file!');
    else res.send('Callback: ' + data);
  });
});

// Promise method
app.get('/read-promise', (req, res) => {
  fsPromises.readFile('sample.txt', 'utf8')
    .then(data => res.send('Promise: ' + data))
    .catch(() => res.send('Error reading file!'));
});

// Async/Await method
app.get('/read-async', async (req, res) => {
  try {
    const data = await fsPromises.readFile('sample.txt', 'utf8');
    res.send('Async/Await: ' + data);
  } catch {
    res.send('Error reading file!');
  }
});

app.listen(PORT, () => console.log(`Server running at http://localhost:${PORT}`));

  <!-- ng-model: Two-way binding with input -->
  <label>Enter your name: </label>
  <input type="text" ng-model="name" placeholder="Your name here" />
  <p>Hello, <span ng-bind="name"></span>!</p>

  <hr>

  <!-- ng-show/ng-hide: Show or hide based on condition -->
  <button ng-click="showList = !showList">
    {{ showList ? 'Hide' : 'Show' }} Fruits List
  </button>

  <ul ng-show="showList">
    <!-- ng-repeat: Loop over array items -->
    <li ng-repeat="fruit in fruits track by $index">
      {{ fruit }}
      <!-- ng-click: Remove item on click -->
      <button ng-click="removeFruit($index)">Remove</button>
    </li>
  </ul>

  <hr>

  <!-- Add new fruit -->
  <input type="text" ng-model="newFruit" placeholder="Add a fruit" />
  <button ng-click="addFruit()">Add Fruit</button>

</body>

<script>
  angular.module("myApp", [])
    .controller("MainCtrl", function($scope) {
      $scope.name = "";

      $scope.showList = true;

      $scope.fruits = ["Apple", "Banana", "Orange"];

      $scope.addFruit = function() {
        if ($scope.newFruit) {
          $scope.fruits.push($scope.newFruit);
          $scope.newFruit = "";
        }
      };

      $scope.removeFruit = function(index) {
        $scope.fruits.splice(index, 1);
      };
    });
</script>

</html>

<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="utf-8">
  <title>AngularJS Router with Guards</title>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular-route.min.js"></script>
</head>
<body ng-controller="NavController">

  <nav>
    <a href="#!/home">Home</a>
    <a href="#!/about">About</a>
    <a href="#!/dashboard">Dashboard</a>
    <span style="float:right">
      <button ng-if="!isAuthenticated" ng-click="login()">Login</button>
      <button ng-if="isAuthenticated" ng-click="logout()">Logout</button>
    </span>
  </nav>
  <hr>

  <div ng-view></div>

  <script>
    var app = angular.module('myApp', ['ngRoute']);

    app.config(function($routeProvider) {
      $routeProvider
        .when('/home', {
          template: '<h2>Home Page</h2><p>Welcome to the Home page.</p>'
        })
        .when('/about', {
          template: '<h2>About Page</h2><p>This is a public About page.</p>'
        })
        .when('/dashboard', {
          template: '<h2>Dashboard</h2><p>Private content. Login required.</p>',
          resolve: {
            auth: function(AuthService, $location) {
              if (!AuthService.isLoggedIn()) {
                alert("You must login first!");
                $location.path('/home');
              }
            }
          }
        })
        .otherwise({ redirectTo: '/home' });
    });

    app.factory('AuthService', function() {
      var loggedIn = false;
      return {
        login: function() { loggedIn = true; },
        logout: function() { loggedIn = false; },
        isLoggedIn: function() { return loggedIn; }
      };
    });

    app.controller('NavController', function($scope, AuthService) {
      $scope.isAuthenticated = AuthService.isLoggedIn();

      $scope.login = function() {
        AuthService.login();
        $scope.isAuthenticated = true;
      };

      $scope.logout = function() {
        AuthService.logout();
        $scope.isAuthenticated = false;
      };
    });
  </script>
</body>
</html>

// server.js
const express = require("express");
const bodyParser = require("body-parser");
const mysql = require("mysql2");

const app = express();
const port = 3000;

// Middleware
app.use(bodyParser.json());

// MySQL Connection
const db = mysql.createConnection({
  host: "localhost",
  user: "root",
  password: "root",
  database: "crud_api"
});

db.connect(err => {
  if (err) throw err;
  console.log(" MySQL Connected...");
});

// Create User
app.post("/users", (req, res) => {
  const { name, email } = req.body;
  const sql = "INSERT INTO users (name, email) VALUES (?, ?)";
  db.query(sql, [name, email], (err, result) => {
    if (err) throw err;
    res.send({ message: "User created", id: result.insertId });
  });
});

// Read All Users
app.get("/users", (req, res) => {
  const sql = "SELECT * FROM users";
  db.query(sql, (err, results) => {
    if (err) throw err;
    res.send(results);
  });
});

// Read Single User
app.get("/users/:id", (req, res) => {
  const { id } = req.params;
  const sql = "SELECT * FROM users WHERE id = ?";
  db.query(sql, [id], (err, result) => {
    if (err) throw err;
    res.send(result[0] || { message: "User not found" });
  });
});

// Update User
app.put("/users/:id", (req, res) => {
  const { id } = req.params;
  const { name, email } = req.body;
  const sql = "UPDATE users SET name = ?, email = ? WHERE id = ?";
  db.query(sql, [name, email, id], (err, result) => {
    if (err) throw err;
    res.send({ message: "User updated" });
  });
});

// Delete User
app.delete("/users/:id", (req, res) => {
  const { id } = req.params;
  const sql = "DELETE FROM users WHERE id = ?";
  db.query(sql, [id], (err, result) => {
    if (err) throw err;
    res.send({ message: "User deleted" });
  });
});

// Start Server
app.listen(port, () => {
  console.log(`Server running at http://localhost:${port}`);
});
